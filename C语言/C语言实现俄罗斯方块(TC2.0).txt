int TimerCounter=0;
void interrupt newtimer(void)
{
	(*oldtimer)();
	TimerCounter++;
}
void SetTimer(void interrupt(*IntProc)(void))
{
	oldtimer=getvect(TIMER);
	disable();
	setvect(TIMER,IntProc);
	enable();
}
void KillTimer()
{
	disable();
	setvect(TIMER,oldtimer);
	enable();
}
